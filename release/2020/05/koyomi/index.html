	end, _ := koyomi.DateFrom("2020-05-31")

		koyomi.WithCalendarID(koyomi.Holiday, koyomi.SolarTerm),

		koyomi.WithEndDate(end),
	).Get()

		return

	b, err := k.EncodeCSV()
	if err != nil {
		fmt.Fprintln(os.Stderr, err)
		return
	}
	io.Copy(os.Stdout, bytes.NewReader(b))
}
